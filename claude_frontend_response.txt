        function renderTerraformCode(code) {
            const lines = code.split('\n');
            const output = document.getElementById('terraformOutput');
            output.innerHTML = '';
            
            lines.forEach((line, index) => {
                const lineEl = document.createElement('div');
                lineEl.className = 'code-line';
                
                let formattedLine = line
                    .replace(/(#.*$)/g, '<span class="code-comment">$1</span>')
                    .replace(/\b(terraform|provider|resource|variable|data|output|module|locals)\b/g, '<span class="code-keyword">$1</span>')
                    .replace(/\b(required_providers|aws|region|cidr_block|tags|name|count|vpc_id)\b(?=\s*=)/g, '<span class="code-property">$1</span>')
                    .replace(/"([^"]*)"/g, '<span class="code-string">"$1"</span>')
                    .replace(/\b(true|false|null)\b/g, '<span class="code-value">$1</span>')
                    .replace(/\b(\d+)\b/g, '<span class="code-value">$1</span>');
                
                lineEl.innerHTML = `<span class="line-number">${index + 1}</span><span class="code-content">${formattedLine || '&nbsp;'}</span>`;
                output.appendChild(lineEl);
            });
        }

        function animateValue(id, start, end, duration) {
            const el = document.getElementById(id);
            const range = end - start;
            const increment = end > start ? 1 : -1;
            const stepTime = Math.abs(Math.floor(duration / range));
            let current = start;
            
            const timer = setInterval(() => {
                current += increment;
                el.textContent = current;
                if (current === end) clearInterval(timer);
            }, stepTime);
        }
